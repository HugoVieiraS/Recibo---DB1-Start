create database recibosdb1;

create table recibo(
idRecibo int auto_increment not null,
sequencia int auto_increment not null,
valor numeric (10,2) not null,
data date,
assinatura varchar (50),
referente varchar (100) not null,

constraint pk_recibo primary key (idRecibo)

);


create table endereco(
idEnd int auto_increment not null,
rua varchar(100) not null,
nomecomple varchar (100) not null,
cep numeric(8) not null,

constraint pk_endereco primary key (idEnd)
);


create table cidade(
idCidade int auto_increment not null,
nomecidade varchar(50) not null,
sigla char (2) not null,

constraint pk_cidade primary key (idCidade)
);


create table usuario(
idUser int auto_increment not null,
nomelogin varchar (20) not null,
cpf numeric(11) not null,
senha md5() not null,

constraint pk_user primary key (idUser)
);

create table cliente(
idCli int auto_increment not null,
nome varchar(100) not null,
cpf numeric (11) not null,

contraint pk_cliente primary key (idCli)
);
